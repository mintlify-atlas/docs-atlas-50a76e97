## Overview

Issuance profiles define reusable templates for certificate issuance. They control certificate validity periods, key usage constraints, subject handling, and cryptographic requirements.

## List Issuance Profiles

<api method="GET" endpoint="/api/ca/v1/profiles" />

Retrieve a paginated list of all issuance profiles.

### Query Parameters

<ParamField query="page_size" type="integer" default="25">
  Number of results per page
</ParamField>

<ParamField query="bookmark" type="string">
  Pagination cursor from previous response
</ParamField>

<ParamField query="sort_by" type="string">
  Field to sort results by
</ParamField>

<ParamField query="sort_mode" type="string">
  Sort order: `asc` or `desc`
</ParamField>

<ParamField query="filter" type="string">
  Filter expression (e.g., `sign_as_ca[eq]=true`)
</ParamField>

### Response

<ResponseField name="list" type="array">
  Array of issuance profile objects
  
  <ResponseField name="id" type="string">
    Profile identifier
  </ResponseField>
  
  <ResponseField name="name" type="string">
    Profile name
  </ResponseField>
  
  <ResponseField name="description" type="string">
    Profile description
  </ResponseField>
  
  <ResponseField name="validity" type="object">
    Certificate validity configuration
    
    <ResponseField name="type" type="string">
      Validity type: `Duration` or `Time`
    </ResponseField>
    
    <ResponseField name="duration" type="string">
      Human-readable duration (e.g., `1y`, `90d`)
    </ResponseField>
  </ResponseField>
  
  <ResponseField name="sign_as_ca" type="boolean">
    Whether certificates should be marked as CA certificates
  </ResponseField>
  
  <ResponseField name="key_usage" type="object">
    X.509 key usage flags
  </ResponseField>
  
  <ResponseField name="extended_key_usages" type="array">
    Array of extended key usage OIDs
  </ResponseField>
  
  <ResponseField name="honor_key_usage" type="boolean">
    Whether to honor key usage from CSR
  </ResponseField>
  
  <ResponseField name="honor_extended_key_usages" type="boolean">
    Whether to honor extended key usages from CSR
  </ResponseField>
  
  <ResponseField name="honor_subject" type="boolean">
    Whether to honor subject from CSR
  </ResponseField>
  
  <ResponseField name="subject" type="object">
    Default subject values to use if not honoring CSR
  </ResponseField>
  
  <ResponseField name="honor_extensions" type="boolean">
    Whether to honor extensions from CSR
  </ResponseField>
  
  <ResponseField name="crypto_enforcement" type="object">
    Cryptographic algorithm enforcement settings
    
    <ResponseField name="enabled" type="boolean">
      Whether crypto enforcement is enabled
    </ResponseField>
    
    <ResponseField name="allow_rsa_keys" type="boolean">
      Whether RSA keys are allowed
    </ResponseField>
    
    <ResponseField name="allowed_rsa_key_sizes" type="array">
      Allowed RSA key sizes (e.g., [2048, 4096])
    </ResponseField>
    
    <ResponseField name="allow_ecdsa_keys" type="boolean">
      Whether ECDSA keys are allowed
    </ResponseField>
    
    <ResponseField name="allowed_ecdsa_key_sizes" type="array">
      Allowed ECDSA key sizes (e.g., [256, 384])
    </ResponseField>
  </ResponseField>
</ResponseField>

<ResponseField name="next" type="string">
  Pagination bookmark for next page
</ResponseField>

### Example Request

## Create Issuance Profile

<api method="POST" endpoint="/api/ca/v1/profiles" />

Create a new issuance profile.

### Request Body

<ParamField body="id" type="string" required>
  Unique profile identifier
</ParamField>

<ParamField body="name" type="string" required>
  Profile name
</ParamField>

<ParamField body="description" type="string">
  Profile description
</ParamField>

<ParamField body="validity" type="object" required>
  Certificate validity configuration
  
  <ParamField body="type" type="string" required>
    Validity type: `Duration` or `Time`
  </ParamField>
  
  <ParamField body="duration" type="string">
    Human-readable duration (e.g., `1y`, `90d`) - required if type is `Duration`
  </ParamField>
  
  <ParamField body="time" type="string">
    Expiration timestamp (ISO 8601) - required if type is `Time`
  </ParamField>
</ParamField>

<ParamField body="sign_as_ca" type="boolean" default={false}>
  Whether to sign certificates as CA certificates
</ParamField>

<ParamField body="honor_key_usage" type="boolean" default={false}>
  Whether to honor key usage from CSR
</ParamField>

<ParamField body="key_usage" type="object">
  X.509 key usage flags (if not honoring CSR)
  
  Example:
  ```json
  {
    "digital_signature": true,
    "key_encipherment": true,
    "data_encipherment": false
  }
  ```
</ParamField>

<ParamField body="honor_extended_key_usages" type="boolean" default={false}>
  Whether to honor extended key usages from CSR
</ParamField>

<ParamField body="extended_key_usages" type="array">
  Array of extended key usage OIDs (if not honoring CSR)
  
  Common values:
  - `1.3.6.1.5.5.7.3.1` - TLS Web Server Authentication
  - `1.3.6.1.5.5.7.3.2` - TLS Web Client Authentication
  - `1.3.6.1.5.5.7.3.3` - Code Signing
</ParamField>

<ParamField body="honor_subject" type="boolean" default={true}>
  Whether to honor subject from CSR
</ParamField>

<ParamField body="subject" type="object">
  Default subject values (if not honoring CSR)
  
  <ParamField body="common_name" type="string">
    Common name (CN)
  </ParamField>
  
  <ParamField body="organization" type="string">
    Organization (O)
  </ParamField>
  
  <ParamField body="organization_unit" type="string">
    Organizational unit (OU)
  </ParamField>
  
  <ParamField body="country" type="string">
    Country (C)
  </ParamField>
  
  <ParamField body="state" type="string">
    State or province (ST)
  </ParamField>
  
  <ParamField body="locality" type="string">
    Locality or city (L)
  </ParamField>
</ParamField>

<ParamField body="honor_extensions" type="boolean" default={false}>
  Whether to honor extensions from CSR
</ParamField>

<ParamField body="crypto_enforcement" type="object">
  Cryptographic algorithm enforcement
  
  <ParamField body="enabled" type="boolean" default={false}>
    Enable crypto enforcement
  </ParamField>
  
  <ParamField body="allow_rsa_keys" type="boolean" default={true}>
    Allow RSA keys
  </ParamField>
  
  <ParamField body="allowed_rsa_key_sizes" type="array">
    Allowed RSA key sizes in bits (e.g., [2048, 4096])
  </ParamField>
  
  <ParamField body="allow_ecdsa_keys" type="boolean" default={true}>
    Allow ECDSA keys
  </ParamField>
  
  <ParamField body="allowed_ecdsa_key_sizes" type="array">
    Allowed ECDSA key sizes in bits (e.g., [256, 384, 521])
  </ParamField>
</ParamField>

### Response

Returns the created issuance profile object.

### Example Request

```bash
curl -X POST "https://your-domain.com/api/ca/v1/profiles" \
  -H "Authorization: Bearer <token>" \
  -H "Content-Type: application/json" \
  -d '{
    "id": "web-server-profile",
    "name": "Web Server Certificate Profile",
    "description": "Profile for TLS server certificates",
    "validity": {
      "type": "Duration",
      "duration": "90d"
    },
    "sign_as_ca": false,
    "honor_key_usage": false,
    "key_usage": {
      "digital_signature": true,
      "key_encipherment": true
    },
    "honor_extended_key_usages": false,
    "extended_key_usages": [
      "1.3.6.1.5.5.7.3.1"
    ],
    "honor_subject": true,
    "honor_extensions": false,
    "crypto_enforcement": {
      "enabled": true,
      "allow_rsa_keys": true,
      "allowed_rsa_key_sizes": [2048, 4096],
      "allow_ecdsa_keys": true,
      "allowed_ecdsa_key_sizes": [256, 384]
    }
  }'
```

## Profile Configuration Examples

### CA Certificate Profile

For issuing intermediate CA certificates:

```json
{
  "id": "intermediate-ca-profile",
  "name": "Intermediate CA Profile",
  "validity": {
    "type": "Duration",
    "duration": "5y"
  },
  "sign_as_ca": true,
  "honor_key_usage": false,
  "key_usage": {
    "cert_sign": true,
    "crl_sign": true
  },
  "honor_subject": false,
  "subject": {
    "organization": "Lamassu IoT",
    "country": "US"
  }
}
```

### IoT Device Profile

For short-lived device certificates:

```json
{
  "id": "iot-device-short",
  "name": "IoT Device (30 days)",
  "validity": {
    "type": "Duration",
    "duration": "30d"
  },
  "sign_as_ca": false,
  "honor_subject": true,
  "crypto_enforcement": {
    "enabled": true,
    "allow_rsa_keys": true,
    "allowed_rsa_key_sizes": [2048],
    "allow_ecdsa_keys": true,
    "allowed_ecdsa_key_sizes": [256]
  }
}
```
